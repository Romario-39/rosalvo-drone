🔗 Conexão com Google Sheets
Como configurar (só uma vez no computador):

1. Acesse sheets.google.com e crie uma planilha nova
2. Clique em Extensões → Apps Script
3. Apague o que estiver lá e cole o código abaixo
4. Clique em Implantar → Nova implantação
5. Tipo: App da Web · Acesso: Qualquer pessoa
6. Copie o link gerado e cole aqui embaixo
function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['Data','Cliente','Fazenda','Período','Serviço','Hectares','Valor/ha','Bruto','Status','Quem Arrumou','Com.Arruma','Piloto','Ha Normal','Ha Feriado','Com.Piloto','Total Com.','Líquido Rosalvo','Observações']);
  }
  sheet.appendRow([data.data,data.cliente,data.fazenda,data.periodo,data.tipo,data.hectares,data.valorHa,data.bruto,data.status,data.arrumaNome,data.comissaoArruma,data.pilotoNome,data.haNormal,data.haFeriado,data.comissaoPiloto,data.totalComissoes,data.liquidoRosalvo,data.obs]);
  return ContentService.createTextOutput('OK');
}
